  function myRight(Str: PChar; var Count: integer): PChar; cdecl; export;
  begin
     if not Assigned(Str) then
     	Result := Nil
     else
     begin
         Result := Str;
         StrPCopy(Str, Copy(Str, Length(Str) - Count + 1, Count));
     end;
  end;
